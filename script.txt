Sub SyncHyperlinks()
    ' This macro creates the "Hand Pointer" links in Column D based on Column G
    Dim ws As Worksheet
    Dim cell As Range
    Set ws = ActiveSheet
    
    On Error Resume Next
    For Each cell In ws.Range("G3:G500")
        Dim dCell As Range
        Set dCell = ws.Cells(cell.Row, "D")
        
        If IsNumeric(cell.Value) And cell.Value <> "" Then
            ' Create a real hyperlink object to force the Hand Pointer
            ws.Hyperlinks.Add Anchor:=dCell, Address:="", SubAddress:="'" & ws.Name & "'!D" & cell.Row
            dCell.Font.Color = RGB(0, 0, 255)
            dCell.Font.Underline = xlUnderlineStyleSingle
        Else
            dCell.Hyperlinks.Delete
            dCell.Font.ColorIndex = xlAutomatic
            dCell.Font.Underline = xlUnderlineStyleNone
        End If
    Next cell
    MsgBox "Hand pointers synchronized!", vbInformation
End Sub
