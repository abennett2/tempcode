Private Sub Worksheet_Change(ByVal Target As Range)
    ' 1. MONITOR COLUMN G: This adds/removes the "Hand Icon"
    If Not Intersect(Target, Range("G3:G500")) Is Nothing Then
        Dim cell As Range
        For Each cell In Intersect(Target, Range("G3:G500"))
            Dim dCell As Range
            Set dCell = Range("D" & cell.Row)
            
            If IsNumeric(cell.Value) And cell.Value <> "" Then
                ' Add the link (This enables the Hand Pointer)
                ActiveSheet.Hyperlinks.Add Anchor:=dCell, Address:="", _
                    SubAddress:="'" & Me.Name & "'!D" & cell.Row
                dCell.Font.Color = RGB(0, 0, 255)
                dCell.Font.Underline = xlUnderlineStyleSingle
            Else
                ' Remove the link (This kills the Hand Pointer)
                dCell.Hyperlinks.Delete
                dCell.Font.ColorIndex = xlAutomatic
                dCell.Font.Underline = xlUnderlineStyleNone
            End If
        Next cell
    End If
End Sub

Private Sub Worksheet_FollowHyperlink(ByVal Target As Hyperlink)
    ' 2. CLICK ACTION: Opens the PowerPoint to the specific slide
    If Not Intersect(Target.Range, Range("D3:D500")) Is Nothing Then
        Call OpenPowerPoint(Target.Range)
    End If
End Sub

Sub OpenPowerPoint(TargetCell As Range)
    Dim pptApp As Object, pptPres As Object
    Dim fileName As String, fullPath As String
    Dim slideNum As Long
    
    slideNum = CLng(TargetCell.Offset(0, 3).Value) ' Looking at G
    fileName = "Charter Dev Form_v1.ppsx"
    fullPath = ThisWorkbook.Path & "\" & fileName
    
    On Error Resume Next
    Set pptApp = GetObject(Class:="PowerPoint.Application")
    If pptApp Is Nothing Then
        Set pptApp = CreateObject(Class:="PowerPoint.Application")
    End If
    
    Set pptPres = pptApp.Presentations(fileName)
    If pptPres Is Nothing Then
        Set pptPres = pptApp.Presentations.Open(fullPath)
    End If
    On Error GoTo 0
    
    If Not pptPres Is Nothing Then
        pptApp.Visible = True
        If pptPres.SlideShowWindow Is Nothing Then
            pptPres.Slides(slideNum).Select
        Else
            pptPres.SlideShowWindow.View.GotoSlide slideNum
        End If
        AppActivate "PowerPoint"
    End If
End Sub
