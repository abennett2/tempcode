Private Sub Worksheet_Change(ByVal Target As Range)
    Dim cell As Range
    ' 1. Only run if a change happens in your Slide Number range (E6:E13)
    If Not Intersect(Target, Range("E6:E13")) Is Nothing Then
        For Each cell In Intersect(Target, Range("E6:E13"))
            ' 2. If the slide number cell is NOT empty and IS a number
            If cell.Value <> "" And IsNumeric(cell.Value) Then
                ' Format the cell to the LEFT (Column D) as a link
                With cell.Offset(0, -1).Font
                    .Color = -16776961 ' Standard Blue
                    .Underline = xlUnderlineStyleSingle
                End With
            Else
                ' 3. Otherwise, reset the cell to the LEFT to normal text
                With cell.Offset(0, -1).Font
                    .ColorIndex = xlAutomatic
                    .Underline = xlUnderlineStyleNone
                End With
            End If
        Next cell
    End If
End Sub
