Private Sub Worksheet_FollowHyperlink(ByVal Target As Hyperlink)
    MsgBox "Event fired! Target range: " & Target.Range.Address & vbCrLf & _
           "SubAddress: " & Target.SubAddress
    If Not Intersect(Target.Range, Me.Range("E3:E500")) Is Nothing Then
        Dim pptApp As Object, pptPres As Object
        Dim fileName As String: fileName = "Charter Dev Form_v1.ppsx"
        Dim fullPath As String: fullPath = ThisWorkbook.Path & "\" & fileName
        Dim slideNum As Long: slideNum = Val(Target.Range.Offset(0, -1).Value)  ' Column D
        MsgBox fullPath

        If slideNum < 1 Then Exit Sub
        
        On Error Resume Next
        Set pptApp = GetObject(, "PowerPoint.Application")  ' Existing PPT instance
        On Error GoTo 0
        
        If pptApp Is Nothing Then
            Set pptApp = CreateObject("PowerPoint.Application")
            pptApp.Visible = True
        End If
        
        ' Check if this specific presentation is already open
        Dim pres As Object
        For Each pres In pptApp.Presentations
            If LCase(pres.Name) = LCase(fileName) Then
                Set pptPres = pres
                Exit For
            End If
        Next
        
        If pptPres Is Nothing Then
            ' Not open â†’ open it
            Set pptPres = pptApp.Presentations.Open(fullPath)
        End If
        
        ' Activate PPT and go to slide
        pptApp.Visible = True
        pptApp.Activate
        
        If pptPres.SlideShowWindow Is Nothing Then
            ' Start slideshow if not running
            pptPres.SlideShowSettings.Run
        End If
        
        pptPres.SlideShowWindow.View.GotoSlide slideNum
        
        ' Optional: Bring to front more reliably
        AppActivate pptApp.Caption
        
        On Error GoTo 0
    End If
End Sub
