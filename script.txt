<div id="associated-items-print-container">
  <p>Loading associated items...</p>
</div>

<script>
(function() {
  var SITE_URL = "https://triwesthcac.sharepoint.com/teams/Facilities-DistributionServices";
  var CHILD_LIST_GUID = "cc4d9f45-eaf3-4168-993d-9344daa7e850";
  var CHILD_PRINT_TEMPLATE = "Default";
  var CHILD_LIST_NAME = "Application Details";

  // Get parent item ID from the URL
  var parentId = GetUrlKeyValue("ID");

  if (!parentId) {
    document.getElementById("associated-items-print-container").innerHTML =
      "<p>Could not determine parent item ID.</p>";
    return;
  }

  // Query child items linked to this parent
  var queryUrl = SITE_URL +
    "/_api/web/lists(guid'" + CHILD_LIST_GUID + "')/items" +
    "?$filter=ParentIDId eq " + parentId +
    "&$select=ID";

  fetch(queryUrl, {
    headers: { "Accept": "application/json;odata=verbose" }
  })
  .then(function(r) { return r.json(); })
  .then(function(data) {
    var items = data.d.results;
    var container = document.getElementById("associated-items-print-container");

    if (!items.length) {
      container.innerHTML = "<p>No associated items found.</p>";
      return;
    }

    container.innerHTML = "";

    items.forEach(function(item) {
      // Build URL to child item's Default print template
      var printUrl = SITE_URL +
        "/_layouts/15/Infowise.ULTIFORMSxx/PrintTemplate.aspx" +
        "?List=" + encodeURIComponent(CHILD_LIST_NAME) +
        "&ID=" + item.ID +
        "&Template=" + encodeURIComponent(CHILD_PRINT_TEMPLATE);

      var wrapper = document.createElement("div");
      wrapper.style.marginBottom = "20px";

      var iframe = document.createElement("iframe");
      iframe.src = printUrl;
      iframe.style.width = "100%";
      iframe.style.border = "none";
      iframe.scrolling = "no";
      iframe.onload = function() {
        try {
          iframe.style.height =
            iframe.contentDocument.body.scrollHeight + "px";
        } catch(e) {
          iframe.style.height = "400px";
        }
      };

      wrapper.appendChild(iframe);
      container.appendChild(wrapper);
    });
  })
  .catch(function(err) {
    document.getElementById("associated-items-print-container").innerHTML =
      "<p>Error loading associated items: " + err + "</p>";
  });
})();
</script>
